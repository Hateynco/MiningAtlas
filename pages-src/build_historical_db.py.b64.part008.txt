def import_bgs_exports(
    storage: Dict[Tuple[str, int, int, str], dict],
    features: List[dict],
    material_ids: Dict[str, int],
    materials_by_bgs_name: Dict[str, List[str]],
) -> None:
    for feature in features:
        props = feature.get("properties", {})
        bgs_name = props.get("bgs_commodity_trans")
        material_names = materials_by_bgs_name.get(bgs_name, [])
        if not material_names:
            continue
        iso2 = props.get("country_iso2_code")
        quantity = props.get("quantity")
        year_text = props.get("year", "")[:4]
        if not iso2 or quantity is None or not year_text.isdigit():
            continue
        year = int(year_text)
        unit = props.get("units") or ""
        note = props.get("concat_figure_notes_text") or props.get("concat_table_notes_text") or ""
        for material_name in material_names:
            add_export_record(
                storage,
                iso2,
                material_ids[material_name],
                year,
                float(quantity),
                unit,
                "bgs-exports",
                note=note,
            )


def row_matches_prefixes(value: str, prefixes: Iterable[str], exclusions: Iterable[str] = ()) -> bool:
    text = (value or "").strip().lower()
    for item in exclusions:
        if item in text:
            return False
    return any(text.startswith(prefix.lower()) for prefix in prefixes)
