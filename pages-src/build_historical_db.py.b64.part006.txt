    "filter": filter_text,
        }
        url = f"{BGS_ITEMS_URL}?{urllib.parse.urlencode(params, quote_via=urllib.parse.quote)}"
        payload = json.loads(fetch_bytes(url).decode("utf-8"))
        chunk = payload.get("features", [])
        features.extend(chunk)
        returned = payload.get("numberReturned", len(chunk))
        matched = payload.get("numberMatched", len(features))
        offset += returned
        if returned == 0 or offset >= matched:
            break

    cache_path.write_text(json.dumps(features, ensure_ascii=False), encoding="utf-8")
    return features


def build_filter(stat_type: str, commodity_names: Iterable[str]) -> str:
    parts = []
    for name in commodity_names:
        safe_name = name.replace("'", "''")
        parts.append(f"bgs_commodity_trans = '{safe_name}'")
    return f"bgs_statistic_type_trans = '{stat_type}' AND ({' OR '.join(parts)})"

def build_country_catalog(local_seed_rows: List[dict]) -> Dict[str, dict]:
    countries: Dict[str, dict] = {}
    for row in local_seed_rows:
        iso2 = row.get("iso2")
        if not iso2:
            continue
        countries[iso2] = {
            "iso2": iso2,
            "iso3": row.get("iso3"),
            "name": row.get("country_name"),
            "lat": row.get("lat"),
            "lng": row.get("lng"),
        }
    return countries


def enrich_countries_with_bgs(countries: Dict[str, dict], features: Iterable[dict]) -> None:
    for feature in features:
        props = feature.get("properties", {})
        iso2 = props.get("country_iso2_code")
        if not iso2:
            continue
        if iso2 not in countries:
            countries[iso2] = {
                "iso2": iso2,
                "iso3": props.get("country_iso3_code"),
                "name": props.get("country_trans"),
                "lat": props.get("pole_of_inaccessibility_latitude"),
                "lng": props.get("pole_of_inaccessibility_longitude"),
            }
        else:
            current = countries[iso2]
            current["iso3"] = current.get("iso3") or props.get("country_iso3_code")
            current["lat"] = current.get("lat") or props.get("pole_of_inaccessibility_latitude")
            current["lng"] = current.get("lng") or props.get("pole_of_inaccessibility_longitude")


def build_country_lookup(countries: Dict[str, dict]) -> Dict[str, str]:
    lookup: Dict[str, str] = {}
    for iso2, country in countries.items():
        if country.get("name"):
            lookup[normalize_name(country["name"])] = iso2
    for alias, iso2 in COUNTRY_ALIASES.items():
        lookup[alias] = iso2
    return lookup
