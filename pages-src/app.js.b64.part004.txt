function groupByUnit(rows, quantityResolver) {
  const grouped = new Map();
  rows.forEach((row) => {
    const unit = row.unit || (materialById(row.material_id)?.unit ?? "");
    const nextValue = (grouped.get(unit) || 0) + quantityResolver(row);
    grouped.set(unit, nextValue);
  });
  return Array.from(grouped.entries())
    .sort((a, b) => b[1] - a[1])
    .map(([unit, quantity]) => ({ unit, quantity }));
}

function syncInputs(primary, secondary, updater) {
  let syncing = false;
  const sync = (source, target) => {
    source.addEventListener("input", (event) => {
      if (syncing) return;
      syncing = true;
      target.value = event.target.value;
      updater(event.target.value);
      syncing = false;
    });
  };
  sync(primary, secondary);
  sync(secondary, primary);
}

function getCoverageLabel(material) {
  if (!material?.coverage) return "";
  const prod = material.coverage;
  const prodLabel = prod.first_production_year
    ? `${prod.first_production_year}-${prod.last_production_year}`
    : "aucune";
  const exportLabel = prod.first_export_year
    ? `${prod.first_export_year}-${prod.last_export_year}`
    : "n.d.";
  return `Production ${prodLabel} | Exports ${exportLabel}`;
}

function normalizeUnitKey(unit) {
  return String(unit || "")
    .toLowerCase()
    .trim()
    .replaceAll(" ", "")
    .replaceAll("(", "")
    .replaceAll(")", "")
    .replaceAll(",", "")
    .replaceAll(".", "")
    .replaceAll("₂", "2")
    .replaceAll("₃", "3")
    .replaceAll("₅", "5")
    .replaceAll("³", "3");
}

function toTonnes(quantity, unit) {
  const key = normalizeUnitKey(unit);
  if (["mt", "mtk2o"].includes(key)) return quantity * 1_000_000;
  if (["kt", "ktlce", "ktreo", "ktwo3", "ktv2o5", "ktu"].includes(key)) return quantity * 1000;
  if (["t", "tonnesmetric", "metricton", "metrictons", "tonnes"].includes(key)) return quantity;
  if (["kilograms", "kg"].includes(key)) return quantity / 1000;
  return null;
}

function toCarats(quantity, unit) {
  const key = normalizeUnitKey(unit);
  if (key === "mct") return quantity * 1_000_000;
  if (key === "carats" || key === "carat") return quantity;
  return null;
}

function toBarrels(quantity, unit) {
  const key = normalizeUnitKey(unit);
  if (key === "mb/j") return quantity * BARRELS_PER_YEAR_FROM_MBD;
  const tonnes = toTonnes(quantity, unit);
  return tonnes == null ? null : tonnes * BARRELS_PER_METRIC_TON_OIL;
}

function toCubicMeters(quantity, unit) {
  const key = normalizeUnitKey(unit);
  if (key === "gm3") return quantity * CUBIC_METERS_PER_GM3;
  return null;
}

function toTonnesU(quantity, unit) {
  const key = normalizeUnitKey(unit);
  if (key === "ktu") return quantity * 1000;
  if (key === "tu") return quantity;
  return null;
}
