  switch (marketUnit) {

    case "carat":

    case "barrel":

function getMaterialBenchmark(materialId) {
  return MARKET_BENCHMARKS[materialId] || null;
}

function estimateValue(materialId, quantity, unit) {
  const benchmark = getMaterialBenchmark(materialId);
  if (!benchmark) return { usd: null, marketQuantity: null, benchmark: null };
  const marketQuantity = quantityToMarketUnits(Number(quantity || 0), unit, benchmark.marketUnit);
  if (!Number.isFinite(marketQuantity)) {
    return { usd: null, marketQuantity: null, benchmark };
  }
  return {
    usd: marketQuantity * benchmark.priceUsd,
    marketQuantity,
    benchmark,
  };
}

function estimateRowsValue(materialId, rows) {
  let usd = 0;
  let marketQuantity = 0;
  let count = 0;
  rows.forEach((row) => {
    const estimate = estimateValue(materialId, row.quantity, row.unit || materialById(materialId)?.unit || "");
    if (!Number.isFinite(estimate.usd) || !Number.isFinite(estimate.marketQuantity)) return;
    usd += estimate.usd;
    marketQuantity += estimate.marketQuantity;
    count += 1;
  });
  return {
    usd: count ? usd : null,
    marketQuantity: count ? marketQuantity : null,
    benchmark: getMaterialBenchmark(materialId),
  };
}

function sumDefined(values) {
  let total = 0;
  let hasValue = false;
  values.forEach((value) => {
    if (!Number.isFinite(value)) return;
    total += value;
    hasValue = true;
  });
  return hasValue ? total : null;
}

function getCountryMaterialEconomics(countryIso2, materialId, year, productionRow = null) {
  const material = materialById(materialId);
  const production = productionRow && material
    ? estimateValue(materialId, productionRow.quantity, material.unit)
    : { usd: null, marketQuantity: null, benchmark: getMaterialBenchmark(materialId) };
  const exportsInfo = getLatestExports(countryIso2, materialId, year);
  const exportsEstimate = estimateRowsValue(materialId, exportsInfo.rows);

  let domesticUsd = null;
  if (
    exportsInfo.year === year &&
    Number.isFinite(production.marketQuantity) &&
    Number.isFinite(exportsEstimate.marketQuantity) &&
    production.benchmark
  ) {
    const domesticQuantity = Math.max(production.marketQuantity - exportsEstimate.marketQuantity, 0);
    domesticUsd = domesticQuantity * production.benchmark.priceUsd;
  }

  return {
    benchmark: production.benchmark || exportsEstimate.benchmark || null,
    production,
    exportsInfo,
    exportsEstimate,
    domesticUsd,
  };
}
