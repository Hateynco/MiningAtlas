rialList.innerHTML = "";
  const search = state.materialSearch.trim().toLowerCase();
  const yearRows = getProductionRows(state.selectedYear);
  const countByMaterial = new Map();
  yearRows.forEach((row) => {
    countByMaterial.set(row.material_id, (countByMaterial.get(row.material_id) || 0) + 1);
  });

  const reset = document.createElement("div");
  reset.className = `mat-item${state.selectedMaterialId == null ? " active" : ""}`;
  reset.innerHTML = `
    <div class="mat-icon" style="background:rgba(240,165,0,.12);border:1px solid rgba(240,165,0,.22)">+</div>
    <div class="mat-info">
      <div class="mat-name">Tous les minerais</div>
      <div class="mat-unit">Vue multi-unites</div>
    </div>
    <div class="mat-count">${countByMaterial.size}</div>
  `;
  reset.addEventListener("click", () => {
    state.selectedMaterialId = null;
    renderAll();
  });
  refs.materialList.appendChild(reset);

  const categories = new Map();
  DB.materials.forEach((material) => {
    if (
      search &&
      !material.name.toLowerCase().includes(search) &&
      !material.category.toLowerCase().includes(search)
    ) {
      return;
    }
    const rows = countByMaterial.get(material.id) || 0;
    const bucket = categories.get(material.category) || [];
    bucket.push({ ...material, rows });
    categories.set(material.category, bucket);
  });

  Array.from(categories.keys()).sort().forEach((category) => {
    const label = document.createElement("div");
    label.className = "cat-label";
    label.textContent = category;
    refs.materialList.appendChild(label);

    categories.get(category).forEach((material) => {
      const item = document.createElement("div");
      item.className = `mat-item${material.id === state.selectedMaterialId ? " active" : ""}`;
      item.innerHTML = `
        <div class="mat-icon" style="background:${material.color}22;border:1px solid ${material.color}44">${escapeHtml(material.icon)}</div>
        <div class="mat-info">
          <div class="mat-name">${escapeHtml(material.name)}</div>
          <div class="mat-unit">${escapeHtml(material.unit)}</div>
        </div>
        <div class="mat-count">${material.rows}</div>
      `;
      item.addEventListener("click", () => {
        state.selectedMaterialId = material.id;
        renderAll();
      });
      refs.materialList.appendChild(item);
    });
  });
}
