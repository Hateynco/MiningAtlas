            );
        DB = {
          years: payload.years || [2022],
          default_year: payload.default_year || 2022,
          materials: payload.materials || [],
          countries,
          production: (payload.production || []).map((row) => ({
            ...row,
            country_iso2: row.country_iso2 || row.iso2,
          })),
          exports: payload.exports || [],
          meta: payload.meta || { notes: [] },
        };
        return;
      }
    } catch (error) {
      console.warn("Load failed", error);
    }
  }
}

function initMap() {
  map = L.map("map", {
    center: [20, 10],
    zoom: 2.5,
    minZoom: 2,
    maxZoom: 8,
    zoomControl: true,
  });

  L.tileLayer("https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png", {
    attribution: "© OpenStreetMap contributors © CARTO",
    subdomains: "abcd",
    maxZoom: 19,
  }).addTo(map);
}

function buildYearSelect() {
  refs.yearSelect.innerHTML = "";
  DB.years.forEach((year) => {
    const option = document.createElement("option");
    option.value = year;
    option.textContent = year;
    refs.yearSelect.appendChild(option);
  });
  state.selectedYear = DB.years.includes(DB.default_year) ? DB.default_year : DB.years[DB.years.length - 1];
  refs.yearSelect.value = String(state.selectedYear);
}

function bindEvents() {
  refs.yearSelect.addEventListener("change", (event) => {
    state.selectedYear = Number(event.target.value);
    cache.summaryByKey.clear();
    renderAll();
  });

  syncInputs(refs.materialSearch, refs.materialSearchRail, (value) => {
    state.materialSearch = value;
    buildMaterialList();
  });

  syncInputs(refs.countrySearch, refs.countrySearchRail, (value) => {
    state.countrySearch = value;
    renderRanking();
  });

  refs.toggleLeft.addEventListener("click", () => {
    state.leftOpen = !state.leftOpen;
    applyPanelState();
  });
  refs.toggleRight.addEventListener("click", () => {
    state.rightOpen = !state.rightOpen;
    applyPanelState();
  });
  refs.toggleLeftOpen.addEventListener("click", () => {
    state.leftOpen = false;
    applyPanelState();
  });
  refs.toggleRightOpen.addEventListener("click", () => {
    state.rightOpen = false;
    applyPanelState();
  });
}

function applyQueryState() {
  const params = new URLSearchParams(window.location.search);
  const year = Number(params.get("year"));
  if (Number.isInteger(year) && DB.years.includes(year)) {
    state.selectedYear = year;
  }

  const materialParam = params.get("material");
  if (materialParam) {
    const matched = DB.materials.find((item) =>
      String(item.id) === materialParam || item.name.toLowerCase() === materialParam.toLowerCase()
    );
    state.selectedMaterialId = matched ? matched.id : null;
  }

  if (params.get("left") === "closed") state.leftOpen = false;
  if (params.get("right") === "closed") state.rightOpen = false;
}
