def make_record_key(country_iso2: str, material_id: int, year: int) -> Tuple[str, int, int]:
    return (country_iso2, material_id, year)


def add_production_record(
    storage: Dict[Tuple[str, int, int], dict],
    country_iso2: str,
    material_id: int,
    year: int,
    quantity: float,
    unit: str,
    source: str,
    priority: int,
    note: Optional[str] = None,
) -> None:
    key = make_record_key(country_iso2, material_id, year)
    if quantity is None or quantity <= 0:
        return
    candidate = {
        "country_iso2": country_iso2,
        "material_id": material_id,
        "year": year,
        "quantity": round(float(quantity), 6),
        "_priority": priority,
    }
    current = storage.get(key)
    if current is None or priority >= current["_priority"]:
        storage[key] = candidate


def add_export_record(
    storage: Dict[Tuple[str, int, int, str], dict],
    country_iso2: str,
    material_id: int,
    year: int,
    quantity: float,
    unit: str,
    source: str,
    note: Optional[str] = None,
) -> None:
    if quantity is None or quantity <= 0:
        return
    key = (country_iso2, material_id, year, unit)
    if key not in storage:
        storage[key] = {
            "country_iso2": country_iso2,
            "material_id": material_id,
            "year": year,
            "quantity": round(float(quantity), 6),
            "unit": unit,
        }
        return
    storage[key]["quantity"] = round(storage[key]["quantity"] + float(quantity), 6)


def import_local_seed(
    storage: Dict[Tuple[str, int, int], dict],
    local_rows: List[dict],
) -> None:
    for row in local_rows:
        add_production_record(
            storage=storage,
            country_iso2=row["iso2"],
            material_id=row["material_id"],
            year=int(row.get("year", 2022)),
            quantity=float(row["quantity"]),
            unit=row["unit"],
            source="local-seed",
            priority=1,
            note=row.get("notes", ""),
        )
