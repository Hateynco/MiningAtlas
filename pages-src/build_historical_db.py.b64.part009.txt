def import_usgs_2026(
    storage: Dict[Tuple[str, int, int], dict],
    material_ids: Dict[str, int],
    materials_meta: Dict[int, dict],
    country_lookup: Dict[str, str],
) -> None:
    with USGS_2026_PATH.open(encoding="cp1252", newline="") as handle:
        reader = csv.DictReader(handle)
        for row in reader:
            commodity = row["Commodity"].replace("\xa0", " ").strip()
            section = row["Section"].strip()
            if not section.lower().startswith("world") or row["Statistics"] != "Production":
                continue
            country_name = row["Country"].replace("\xa0", " ").strip()
            if country_name in {"Other countries", "World total"}:
                continue
            iso2 = country_lookup.get(normalize_name(country_name))
            if not iso2:
                continue
            detail = row["Statistics_detail"]
            for material_name, config in SERIES_CONFIG.items():
                rules = config.get("usgs2026")
                if not rules or commodity != rules["commodity"]:
                    continue
                if not row_matches_prefixes(detail, rules["detail_prefixes"], rules.get("detail_excludes", [])):
                    continue
                year_text = row["Year"].strip()
                if not year_text.isdigit():
                    continue
                quantity = parse_number(row["Value"])
                material_id = material_ids[material_name]
                add_production_record(
                    storage,
                    iso2,
                    material_id,
                    int(year_text),
                    convert_quantity_to_display_unit(quantity or 0, row.get("Unit", ""), materials_meta[material_id]["unit"]),
                    materials_meta[material_id]["unit"],
                    "usgs-mcs2026",
                    priority=5,
                    note=row.get("Notes", ""),
                )
