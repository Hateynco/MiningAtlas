      const exportsInfo = getLatestExports(summary.iso2, row.material_id, state.selectedYear);
      const exportsEstimate = estimateRowsValue(row.material_id, exportsInfo.rows);

      let domesticUsd = null;
      if (
        exportsInfo.year === state.selectedYear &&
        Number.isFinite(production.marketQuantity) &&
        Number.isFinite(exportsEstimate.marketQuantity) &&
        production.benchmark
      ) {
        domesticUsd = Math.max(production.marketQuantity - exportsEstimate.marketQuantity, 0) * production.benchmark.priceUsd;
      }

      return {
        row,
        material,
        production,
        exportsInfo,
        exportsEstimate,
        domesticUsd,
      };
    })
    .sort((a, b) => {
      const aValue = a.production.usd ?? -1;
      const bValue = b.production.usd ?? -1;
      if (aValue !== bValue) return bValue - aValue;
      return (b.row.quantity || 0) - (a.row.quantity || 0);
    });
}

function getLatestExports(countryIso2, materialId, year) {
  const rows = materialId == null
    ? (cache.exportsByCountry.get(countryIso2) || [])
    : (cache.exportsByMaterialCountry.get(`${countryIso2}|${materialId}`) || []);

  let latestYear = null;
  const exact = [];
  rows.forEach((row) => {
    if (row.year === year) {
      exact.push(row);
      latestYear = year;
    }
  });
  if (exact.length) return { year: latestYear, rows: exact };

  const byYear = new Map();
  rows.forEach((row) => {
    if (row.year > year) return;
    const yearRows = byYear.get(row.year) || [];
    yearRows.push(row);
    byYear.set(row.year, yearRows);
  });

  const availableYears = Array.from(byYear.keys()).sort((a, b) => b - a);
  if (!availableYears.length) return { year: null, rows: [] };
  latestYear = availableYears[0];
  return { year: latestYear, rows: byYear.get(latestYear) || [] };
}
