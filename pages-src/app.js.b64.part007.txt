et(row.country_iso2) || {
        iso2: row.country_iso2,
        name: country.name,
        lat: country.lat,
        lng: country.lng,
        score: 0,
        rows: [],
        topRow: null,
        materialCount: 0,
      };
      current.rows.push(row);
      current.materialCount += 1;
      if (!current.topRow || row.quantity > current.topRow.quantity) current.topRow = row;
      current.score += row.quantity / (maxByMaterial.get(row.material_id) || row.quantity || 1);
      summaries.set(row.country_iso2, current);
    });
  }

  const list = Array.from(summaries.values()).sort((a, b) => b.score - a.score);
  cache.summaryByKey.set(cacheKey, list);
  return list;
}
