def build_coverage(materials: List[dict], production: List[dict], exports: List[dict]) -> Dict[int, dict]:
    prod_years: Dict[int, set] = defaultdict(set)
    export_years: Dict[int, set] = defaultdict(set)
    for row in production:
        prod_years[row["material_id"]].add(row["year"])
    for row in exports:
        export_years[row["material_id"]].add(row["year"])

    result: Dict[int, dict] = {}
    for material in materials:
        mid = material["id"]
        prod = sorted(prod_years.get(mid, set()))
        exp = sorted(export_years.get(mid, set()))
        result[mid] = {
            "production_years": prod,
            "export_years": exp,
            "first_production_year": prod[0] if prod else None,
            "last_production_year": prod[-1] if prod else None,
            "first_export_year": exp[0] if exp else None,
            "last_export_year": exp[-1] if exp else None,
        }
    return result
